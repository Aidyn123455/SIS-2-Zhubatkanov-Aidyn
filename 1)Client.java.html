1    package network;
2    
3    import geometry.Circle;
4    import geometry.Rectangle;
5    import geometry.Shape;
6    import java.io.*;
7    import java.net.*;
8    import java.util.Scanner;
9    
10   public class Client {
11       public static void main(String[] args) {
12           try (Socket socket = new Socket("localhost", 5000);
13                ObjectOutputStream output = new ObjectOutputStream(socket.getOutputStream());
14                ObjectInputStream input = new ObjectInputStream(socket.getInputStream());
15                Scanner scanner = new Scanner(System.in)) {
16   
17               while (true) {
18                   System.out.println("Выберите фигуру (круг или прямоугольник) или 'Q' для выхода:");
19                   String choice = scanner.nextLine();
20   
21                   if (choice.equalsIgnoreCase("q")) {
22                       System.out.println("Завершение работы...");
23                       output.writeObject(null);
24                       output.flush();
25                       break;
26                   }
27   
28                   Shape shape = null;
29                   if (choice.equalsIgnoreCase("круг")) {
30                       System.out.print("Введите радиус: ");
31                       double radius = scanner.nextDouble();
32                       shape = new Circle(radius);
33                   } else if (choice.equalsIgnoreCase("прямоугольник")) {
34                       System.out.print("Введите ширину: ");
35                       double width = scanner.nextDouble();
36                       System.out.print("Введите высоту: ");
37                       double height = scanner.nextDouble();
38                       shape = new Rectangle(width, height);
39                   }
40                   scanner.nextLine();
41   
42   
43                   output.writeObject(shape);
44                   output.flush();
45   
46                   // otvet server
47                   String response = (String) input.readObject();
48                   System.out.println(response);
49               }
50           } catch (IOException | ClassNotFoundException e) {
51               e.printStackTrace();
52           }
53       }
54   }
